// --- データ管理 --- // ID生成用のカウンター。初期データで使用されるIDより大きな値で開始 let nextId = 5000; // 初期データ let checklistData = [ { id: nextId++, name: "外出時の必需品", items: [ { id: nextId++, name: "財布", checked: false }, { id: nextId++, name: "スマートフォン", checked: false }, { id: nextId++, name: "鍵", checked: false } ] }, { id: nextId++, name: "仕事・学校", items: [ { id: nextId++, name: "バッグ", checked: false }, { id: nextId++, name: "ノートPC", checked: false }, { id: nextId++, name: "充電器", checked: false }, { id: nextId++, name: "筆記用具", checked: false }, { id: nextId++, name: "書類", checked: false } ] } ]; const categoriesContainer = document.getElementById('categories-container'); const addCategoryBtn = document.getElementById('add-category-btn'); const resetAllBtn = document.getElementById('reset-all'); // --- 永続化 (ローカルストレージ) --- function saveDataToLocalStorage() { localStorage.setItem('checklistData', JSON.stringify(checklistData)); } function loadDataFromLocalStorage() { const storedData = localStorage.getItem('checklistData'); if (storedData) { checklistData = JSON.parse(storedData); // 【重要修正】ローカルストレージから読み込んだ後、nextIdを正確に更新 let maxId = 0; checklistData.forEach(cat => { maxId = Math.max(maxId, cat.id); cat.items.forEach(item => { maxId = Math.max(maxId, item.id); }); }); nextId = maxId + 1; console.log("Next ID set to:", nextId); // デバッグ用 } } // --- DOM操作と描画 --- function createCategoryHTML(category) { const totalItems = category.items.length; const checkedItems = category.items.filter(item => item.checked).length; const progress = totalItems === 0 ? 0 : Math.round((checkedItems / totalItems) * 100); const itemsHTML = category.items.map(item => `
    • `).join(''); return `
    ${category.name}
    進捗 ${progress}%
      ${itemsHTML}
    `; } function renderChecklist() { categoriesContainer.innerHTML = checklistData.map(createCategoryHTML).join(''); saveDataToLocalStorage(); } // --- ロジック関数 --- /** **【最重要修正】カテゴリ追加機能** */ function addCategory(categoryName) { // promptがキャンセルされた場合(null)または空文字列の場合を厳密にチェック if (categoryName === null) { return; // キャンセルの場合は何もしない } const trimmedName = categoryName.trim(); if (trimmedName) { console.log("Adding category:", trimmedName, "with ID:", nextId); // デバッグ用 checklistData.push({ id: nextId++, // カウンタIDを使用 name: trimmedName, items: [] }); renderChecklist(); } else { alert("カテゴリ名を入力してください。"); } } /** アイテムのチェック状態を切り替える */ function toggleItem(categoryId, itemId) { const category = checklistData.find(cat => cat.id === categoryId); if (category) { const item = category.items.find(it => it.id === itemId); if (item) { item.checked = !item.checked; } } renderChecklist(); } /** カテゴリ内のアイテムを追加する */ function addItem(categoryId, itemName) { const category = checklistData.find(cat => cat.id === categoryId); const trimmedName = itemName ? itemName.trim() : ''; if (category && trimmedName) { category.items.push({ id: nextId++, name: trimmedName, checked: false }); renderChecklist(); } else if (itemName !== null) { // promptでキャンセルされていない場合のみアラート alert("アイテム名を入力してください。"); } } /** アイテムを削除する */ function deleteItem(categoryId, itemId) { const category = checklistData.find(cat => cat.id === categoryId); if (category) { category.items = category.items.filter(item => item.id !== itemId); } renderChecklist(); } /** カテゴリを削除する */ function deleteCategory(categoryId) { checklistData = checklistData.filter(cat => cat.id !== categoryId); renderChecklist(); } /** 全てのチェック状態をリセットする */ function resetAllChecks() { checklistData.forEach(category => { category.items.forEach(item => { item.checked = false; }); }); renderChecklist(); } // --- イベントリスナー --- function setupEventListeners() { // カテゴリ追加ボタン addCategoryBtn.addEventListener('click', () => { const newCategoryName = prompt("新しいカテゴリ名を入力してください:"); addCategory(newCategoryName); }); // すべてリセットボタン resetAllBtn.addEventListener('click', () => { if (confirm("全てのチェック状態をリセットしてもよろしいですか？")) { resetAllChecks(); } }); // イベント委譲 categoriesContainer.addEventListener('click', (e) => { const target = e.target; // data-*属性の値は文字列から数値へ変換 const categoryId = parseInt(target.dataset.categoryId); const itemId = parseInt(target.dataset.itemId); if (target.matches('input[type="checkbox"]')) { toggleItem(categoryId, itemId); } else if (target.matches('.delete-item')) { deleteItem(categoryId, itemId); } else if (target.matches('.delete-category')) { if (confirm(`「${target.parentElement.querySelector('.category-title').textContent}」カテゴリを削除してもよろしいですか？`)) { deleteCategory(categoryId); } } else if (target.matches('.add-item-btn')) { const newItemName = prompt("追加するアイテム名を入力してください:"); addItem(categoryId, newItemName); } }); } // --- アプリの初期化 --- document.addEventListener('DOMContentLoaded', () => { loadDataFromLocalStorage(); renderChecklist(); setupEventListeners(); });
